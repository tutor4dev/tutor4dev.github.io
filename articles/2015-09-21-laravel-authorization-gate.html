Implement Laravel Authorization ด้วย Gate

ความสามารถในการทำ Authentication ที่มากับ Laravel แบบ Out-of-the-Box นั้นเพิ่มความสะดวกให้ Developer เป็นอย่างมาก เมื่อใช้ Auth ร่วมกับ Eloquent แล้ว Developer อย่างพวกเราแทบจะไม่ต้องทำอะไรเลยก็สามารถมีระบบ Authentication ให้กับ Web App ของเราได้ทันที

หลังจากจัดการเรื่อง Authentication แล้ว เราก็ยังมีความต้องการที่จะต้องจัดการเรื่อง Access Control List (ACL) เพิ่มเติม แต่หลังจากนี้ไม่มีอะไรยุ่งยากอีกต่อไปเมื่อ Laravel Released 5.1.11 ได้เพิ่มเติม Facade และ ServiceProvider สำหรับการทำ Authorization นั่นคือ Gate

ตรวจสอบ Released Version ของ Laravel 5

พิมพ์คำสั่งที่ Terminal


php artisan --version

การ Upgrade จาก Laravel Released ก่อนหน้า

เนื่องจากการ Upgrade จาก Released ก่อนหน้า 5.1.11 โดยการใช้ composer update นั้น เป็นเพียงการ Update Illuminate เท่านั้น จึงส่งผลให้ไฟล์ที่อยู่นอก Folder vendor ไม่ได้รับการเปลี่ยนแปลงแต่อย่างใด เราจึงต้องทำการ Manual Update เพิ่มเติมดังนี้

เพิ่มไฟล์ app/Providers/AuthServiceProvider.php


namespace App\Providers;

use Illuminate\Contracts\Auth\Access\Gate as GateContract;
use Illuminate\Foundation\Support\Providers\AuthServiceProvider as ServiceProvider;

class AuthServiceProvider extends ServiceProvider
{
    /**
     * The policy mappings for the application.
     *
     * @var array
     */
    protected $policies = [
        'App\Model' => 'App\Policies\ModelPolicy',
    ];

    /**
     * Register any application authentication / authorization services.
     *
     * @param  \Illuminate\Contracts\Auth\Access\Gate  $gate
     * @return void
     */
    public function boot(GateContract $gate)
    {
        parent::registerPolicies($gate);

        //
    }
}

แก้ไขไฟล์ config/app.php


/*
 * Application Service Providers...
 */

App\Providers\AppServiceProvider::class,
App\Providers\AuthServiceProvider::class,

...

'aliases' => [
    ...
    'Gate' => Illuminate\Support\Facades\Gate::class,
    ...
];

การกำหนด Policy

การกำหนด Policy ในการเข้าถึง Resource ต่างๆ ภายใน App นั้น เราจะกำหนดไว้ที่ไฟล์ app/Providers/AuthServiceProvider.php ภายใน Function ที่ชื่อ boot()


class AuthServiceProvider extends ServiceProvider
{
    public function boot(GateContract $gate)
    {
        parent::registerPolicies($gate);

        $gate->define('list-employee', function($user) {
            return $user->role === 'admin';
        });
    }
}

สังเกตุว่า Function boot() จะมีการ Inject GateContract เข้ามาเพื่อใช้งาน

เรากำหนด Policy โดยการใช้ $gate->define และ กำหนดชื่อ เช่น list-employee พร้อมทั้งมีการรับ Parameter $user จากตัวอย่างข้างต้น $user จะเป็นตัวแปรที่เก็บค่า User ของ Session ซึ่งมาจากการเรียกใช้ Auth::attemp() ในการทำ Authentication ซึ่งเราได้กำหนด Policy ไว้ว่า User ที่สามารถเข้าถึง list-employee ได้นั้นจะต้องมี Field role ในฐานข้อมูลเท่ากับ admin

เนื่องจาก Gate รับค่าต่อมาจาก $user ดังนั้นหากยังไม่มีการ Logged in การทำ Authorization ด้วย Gate จะคืนค่า false โดยอัตโนมัติ

ตรวจสอบ Authorization ใน Controller

เราสามารถอ้างถึง Gate Facade ได้ทันทีเมื่อเรามีความต้องการที่จะทำการตรวจสอบสิทธิ์ในการเข้าถึง โดยการใช้ Gate::allows() แล้วตามด้วยชื่อของ Policy ที่ได้กำหนดไว้

Controller


if (\Gate::allows('list-employee')) {
    echo 'You have permission';
} else {
    echo 'You have no permissions';
}

Gate::allows() จะอนุญาตให้ทำเมื่อ Policy คืนค่า false ส่วน Gate::denies() จะให้ผลลัพธ์ตรงกันข้าม

การกำหนด Policy แบบมีการอ้างอิง Model อื่น

ยกตัวอย่างเช่น ในกรณีที่เราจะอนุญาตให้ User สามารถลบเฉพาะ Post ของตัวเองได้เท่านั้น

app/Providers/AuthServiceProvider.php


$gate->define('delete-post', function($user, $post) {
    return $post->user_id = $user->id;
});

ตรวจสอบว่า Post ที่ User ต้องการลบมี ID ของเจ้าของ Post ตรงกับ ID ของ User ตาม Session หรือไม่

Controller


$post = \App\Post::find($id);

if (\Gate::allows('delete-post', $post)) {
    echo 'You have permission to delete';
} else {
    echo 'You have no permissions to delete';
}

ตรวจสอบ Authorization ใน Model

เราสามารถทำการตรวจสอบ Authorization ที่ระดับ Model ได้เช่นกัน โดยการใช้ can() ของ App\User ซึ่งจะมีค่าเทียบเท่ากับการใช้ Gate::allows() ในระดับ Controller ส่วน cannot() จะเทียบเท่ากับ Gate::denies()

Model


if ($user->can('delete-post', $post)) {
    // You have permission to delete
}

ตรวจสอบ Authorization ใน Blade

เราสามารถทำการตรวจสอบ Authorization ที่ระดับของ Blade ได้เช่นกัน เช่น เราต้องการจะแสดง Link ในการลบ Post เฉพาะกรณีที่ User มีสิทธิในการเข้าถึงเท่านั้น

Blade


@can('delete-post', $post)
    <input type="submit" value="Delete">
@endcan

บทสรุป

การทำ Authorization นั้นเป็นสิ่งที่มีความจำเป็นแทบในทุกๆ App ซึ่ง Gate ใน Laravel Released 5.1.11 นั้นช่วยให้ชีวิตของ Developer ง่ายยิ่งขึ้นอีกเยอะ ในบทความนี้เป็นเพียงการพูดถึง Gate ในระดับพื้นฐานเท่านั้น หากผู้อ่านสนใจใช้งาน Gate ในระดับสูงสามารถศึกษาได้จาก Documentation ของ Laravel โดยตรงครับ

Code ตัวอย่างในบทความนี้ถูกทดสอบบน Laravel 5.1.x
